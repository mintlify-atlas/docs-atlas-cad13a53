  <p className="text-base dark:text-gray-400 text-gray-600 mb-8">Get up and running with robot navigation in minutes</p>
  
  <Steps>
    <Step title="Set up NVIDIA NIM">
      Pull and run the Cosmos Reason2 NIM container with your NGC API key:
      
      ```bash
      export NGC_API_KEY="your-key-here"
      export LOCAL_NIM_CACHE=~/.cache/nim
      docker run -it --rm --runtime=nvidia --gpus all \
        -e NGC_API_KEY=$NGC_API_KEY \
        -v "$LOCAL_NIM_CACHE:/opt/nim/.cache" \
        -p 8000:8000 \
        nvcr.io/nim/nvidia/cosmos-reason2-2b:1.6.0
      ```
    </Step>
    
    <Step title="Deploy the API bridge">
      Run the FastAPI bridge service to process navigation requests:
      
      ```bash
      export BRIDGE_API_KEY="super-secret-key"
      export NIM_BASE_URL="http://127.0.0.1:8000/v1"
      python3 -m uvicorn app:app --host 0.0.0.0 --port 8080
      ```
    </Step>
    
    <Step title="Make your first navigation request">
      Send an image to get navigation decisions:
      
      ```python
      import requests
      import base64
      
      with open("scene.jpg", "rb") as f:
          image_b64 = base64.b64encode(f.read()).decode()
      
      response = requests.post(
          "http://localhost:8080/v1/reason2/action",
          headers={"X-API-Key": "super-secret-key"},
          json={
              "image_b64": image_b64,
              "instruction": "Check if the path forward is clear"
          }
      )
      
      print(response.json())
      # {"reasoning": "...", "action": {"move": "linear_x", "distance": 0.5}}
      ```
    </Step>
  </Steps>
